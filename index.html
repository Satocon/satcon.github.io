import pandas as pd import folium from pyodide.http import open_url from folium.features import DivIcon from folium.features import CustomIcon sheet_url = ("https://docs.google.com/spreadsheets/d/1bG-512BNgFMSKzmBdVZEBzcyc8QsW3k-kOxe85OZE0U/export?format=csv&gid=0") url_content = open_url(sheet_url) store_lists = pd.read_csv(url_content) m = folium.Map(location=[3.9948833532516526, 20.82226593026007], zoom_start=2.1,maxZoom=2.5) for index, row in store_lists.iterrows(): folium.Marker( location= [row['緯度'],row['経度']], icon=CustomIcon(icon_image=row['アイコン'], icon_size=(20,20), icon_anchor=(10,10), popup_anchor=(0,0)), ).add_to(m) folium.map.Marker( location= [row['緯度'],row['経度']], icon=DivIcon( icon_size=(105,20), icon_anchor=(-8,6), html=row['駅名'],), tooltip=row['カード1']+ "\r\n" + row['カード2']+ "\r\n" + row['カード3']+ "\r\n" + row['カード4']+ "\r\n" + row['カード5']+ "\r\n" + row['カード6']+ "\r\n" + row['カード7']+ "\r\n" + row['カード8'], ).add_to(m) folium.LayerControl().add_to(m) m
