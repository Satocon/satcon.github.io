import pandas as pd import folium from pyodide.http import open_url sheet_url = ("https://docs.google.com/spreadsheets/d/1bG-512BNgFMSKzmBdVZEBzcyc8QsW3k-kOxe85OZE0U/export?format=csv&gid=0") url_content = open_url(sheet_url) shoplists = pd.read_csv(url_content) m = folium.Map(location=[35.70,139.66], zoom_start=3) for index, row in shoplists.iterrows(): folium.Circle( location= [row['緯度'],row['経度']], radius=120000, popup=folium.Popup(row['カード1']+ "\r\n" + row['カード2']+ "\r\n" + row['カード3']+ "\r\n" + row['カード4']+ "\r\n" + row['カード5']+ "\r\n" + row['カード6']+ "\r\n" + row['カード7']+ "\r\n" + row['カード8'], max_width=100), color=row['色'], fill_color=row['色'], fill=True ).add_to(m) folium.LayerControl().add_to(m) m
